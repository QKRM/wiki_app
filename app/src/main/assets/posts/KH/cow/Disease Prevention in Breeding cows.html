ការការពារជំងឺក្នុងគោចិញ្ចឹមសម្រាប់បន្តពូជ

▣ ១. ការការពាររោគស្រ្តី

● វិធានការការពារសំខាន់ៗ៖

  • ការគ្រប់គ្រងអាហារ និងអាហារូបត្ថម្ភត្រឹមត្រូវ
  • ហាត់ប្រាណគ្រប់គ្រាន់ ដើម្បីការពារទាំងការខ្វះអាហារូបត្ថម្ភ និងជំងឺធាត់
  • ការគ្រប់គ្រងអនាម័យមុន និងបន្ទាប់ពីសម្រាល៖
  • ការពារការបំពុលតាមរយៈផ្លូវភេទ
  • គ្រប់គ្រងស្បូននៅសល់ឱ្យទាន់ពេល
  • ត្រួតពិនិត្យ និងព្យាបាលការបញ្ចេញសារធាតុរយៈពេល ២–៣សប្តាហ៍បន្ទាប់ពីសម្រាល

គោលដៅបន្តពូជចុងក្រោយគួរតែមានកូនមួយក្បាលក្នុងមួយឆ្នាំសម្រាប់គោមួយក្បាល។

● រោគស្រ្តីដែលជួបញឹកញាប់៖

១. ការប៉ះពាល់ដល់ពងស្វាស

  • រួមមានការអភិវឌ្ឍមិនគ្រប់គ្រាន់ ការរលាយ ឬការស្លាប់នៃពងស្វាស
  • បណ្តាលមកពីការអភិវឌ្ឍរបស់ផ្ចិតមិនល្អ
  • ការការពារ៖ ផ្តល់អាហារក្រាស់គុណភាពខ្ពស់ ប្រូតេអ៊ីន ផូស្វ័រ និងវីតាមីន A
  • ការព្យាបាល៖ ចាក់ថ្នាំ PMSG ឬ HCG (៥០០–១,០០០μg) តាមសាច់ដុំ

២. រោគបញ្ហាដល់ការបញ្ចេញពង

  • រួមមានរដូវយូរ និងស៊ីស្តិចនៅពងស្វាស
  • បណ្តាលមកពីការមានភាពមិនសមស្របរបស់ LH និង FSH
  • ការព្យាបាល៖ ចាក់ថ្នាំ HCG តាមសាច់ដុំ
  • ប្រភេទប្រេង៖ ៥,០០០–១០,០០០ IU
  • ប្រភេទទឹក៖ ១០,០០០ IU

៣. គោបន្តពូជម្តងហើយម្តងទៀត

  • សរីរាង្គបន្តពូជ និងរង្វិលរដូវធម្មតា ប៉ុន្តែមិនមានផ្ទៃពោះបន្ទាប់ពីព្យាយាម ៣ដងឡើងទៅ
  • មិនមានការព្យាបាលជាក់លាក់ទេ ត្រូវការរួមបញ្ចូលអាហារូបត្ថម្ភ និងអនាម័យត្រឹមត្រូវ

៤. រោគស្បូន

  • បណ្តាលមកពី Vibrio, Brucella, Trichomonas ឬ E. coli
  • បណ្តាលឱ្យមានទឹកនោមពោះស្បូន
  • ការព្យាបាល៖ លាងស្បូន និងចាក់ថ្នាំប្រឆាំងជំងឺក្នុងស្បូន

▣ ២. ការគ្រប់គ្រងសុខភាពទឹកដោះ

ការរក្សាសុខភាពទឹកដោះគឺសំខាន់សម្រាប់ផលិតភាព សុខុមាលភាពគោ និងប្រាក់ចំណេញកសិដ្ឋាន។ សុខភាពទឹកដោះអន់នាំឱ្យ៖ - ការថយចុះផលិតភាពទឹកដោះ - គុណភាពទឹកដោះអន់ - ផ្តាសាយ បាត់បង់ចំណង់អាហារ និងស្លាប់ផងដែរ

● ប្រភេទ និងរោគសញ្ញានៃជំងឺទឹកដោះ៖

១. ជំងឺទឹកដោះបង្ហាញរោគសញ្ញា

  • ទឹកដោះរលាក ក្តៅ និងឈឺ
  • មានការផ្លាស់ប្តូរមើលឃើញក្នុងទឹកដោះ (កក សារធាតុពុល)
  • ករណីធ្ងន់ធ្ងរអាចត្រូវកាត់ចេញ

២. ជំងឺទឹកដោះមិនបង្ហាញរោគសញ្ញា

  • មិនមានរោគសញ្ញាដែលមើលឃើញ
  • រកឃើញតែតាមតេស្តមន្ទីរពិសោធន៍ (CMT, ការចិញ្ចឹមបាក់តេរី)
  • ប្រសិនបើមិនព្យាបាល ជំងឺនេះអាចរីកចម្រើនទៅជាជំងឺទឹកដោះបង្ហាញរោគសញ្ញា និងបណ្តាលឱ្យខាតបង់សេដ្ឋកិច្ច

● កត្តាហានិភ័យសម្រាប់ទឹកដោះអន់៖

  1. កម្រិតសរីរាង្គមិនល្អ៖
  2. ស្វាសទឹកដោះខ្សោយ រាងកាយឬបំពង់ទឹកដោះមិនធម្មតា
  3. កន្លែងចិញ្ចឹមអនាម័យមិនល្អ៖
  4. ទ្រុងកខ្វក់ ខ្វះការសម្អាត កន្លែងដាក់កម្រាលអនាម័យមិនល្អ
  5. អាហារូបត្ថម្ភមិនសមស្រប៖
  6. ខ្វះរ៉ែ/វីតាមីន ផ្លាស់ប្តូរអាហាររហ័ស អាហារចាស់
  7. បញ្ហាហ័រមូន/សមាសធាតុចម្លង៖
  8. អេស្ត្រូសែនច្រើនពេក ឬខ្វះប្រូជេស្ទេរូន
  9. ការថែទាំមុន/បន្ទាប់សម្រាលមិនអនាម័យ៖
  10. មិនបានប្រើថ្នាំសំអាតស្វាសទឹកដោះ
  11. បញ្ហាម៉ាស៊ីនបូមទឹកដោះ៖
  12. សំពាធស្រូបខ្ពស់ ស្បែកស្វាសចាស់ ល្បឿនបូមលឿន (>៦៥ដង/នាទី)
  13. អនាម័យបូមទឹកដោះមិនល្អ៖
  14. ប្រើកន្សែងរួមគ្នា មិនបានលាងស្វាសទឹកដោះ ឧបករណ៍កខ្វក់
  15. បញ្ហាបន្តពូជ៖
  16. ចិញ្ចឹមគោដែលមានភាពងាយរងជំងឺទឹកដោះតាមបន្ទុកពូជ

▣ ៣. វិធានការការពារជំងឺទឹកដោះ

● វិធីបូមទឹកដោះត្រឹមត្រូវ និងការត្រួតពិនិត្យម៉ាស៊ីន

  • ត្រួតពិនិត្យជាប្រចាំ៖
  • សំពាធស្រូប (ប្រសើរ៖ ~៥០ kPa)
  • ល្បឿនបូមទឹកដោះ
  • ការប្តូរស្បែកស្វាស
  • សមត្ថភាពម៉ាស៊ីនបូម

ការអនុវត្តសំខាន់ៗ៖

  1. សម្អាតតែស្វាសទឹកដោះ (មិនមែនទាំងទឹកដោះទេ)
  2. បូមទឹកដោះមុន (Pre-milking stripping):
  3. ជួយបញ្ចេញអុកស៊ីតូស៊ីន
  4. ជួយរកឃើញជំងឺទឹកដោះដំបូង
  5. ការពារទឹកដោះមានថ្នាំប្រឆាំងជំងឺមិនឱ្យចូលរួម

⚠️ ប្រើពែងសម្រាប់សាកល្បង (មិនមែនលើដី) ដើម្បីជៀសវាងការបំពុលបរិស្ថាន

  1. ជៀសវាងបូមទឹកដោះលើសកំណត់៖
  2. ការបូមទឹកដោះមិនគួរលើស ៥នាទី
  3. ប្រើម៉ាស៊ីនបូមច្រើនបំផុត ៣គ្រឿងក្នុងមនុស្សម្នាក់

  4. ជៀសវាងបូមបន្ទាប់ពីទឹកដោះចេញធម្មតា

  5. បូមគោមានជំងឺចុងក្រោយ និងដាក់ឲ្យដាច់ពីគោដទៃ ដើម្បីកាត់បន្ថយការរីករាលដាលជំងឺ

● ការចិញ្ច្រាំស្វាសទឹកដោះ និងការសំអាតដោយស្ព្រេ

  • បន្ទាប់ពីបូមទឹកដោះ បំពង់ស្វាសនៅបើករយៈពេល ១៥–៣០នាទី ដែលមានហានិភ័យឱ្យបាក់តេរីចូល
  • ការសំអាតគឺសំខាន់ ដោយប្រើ៖
  • សារធាតុចិញ្ច្រាំស្វាស៖ ប្តូរទឹកជាប្រចាំដើម្បីរក្សាប្រសិទ្ធភាព
  • ស្ព្រេ៖ ប្រាកដថាបានសំអាតស្វាសទាំងមូល

● ការគ្រប់គ្រងរយៈពេលស្ងួត

ការព្យាបាលគោស្ងួតគឺសំខាន់សម្រាប់សុខភាពទឹកដោះ៖ - ប្រើថ្នាំបន្តពូជឬថ្នាំប្រឆាំងជំងឺក្នុងស្បូនដែលមានប្រសិទ្ធភាពយូរ - ការពារជំងឺទឹកដោះក្នុងរយៈពេលស្ងួត និងដំបូងបំបៅ

សកម្មភាពសំខាន់ៗ៖

  • ប្រមូលគំរូទឹកដោះតាមកន្លែងសម្រាប់តេស្តមន្ទីរពិសោធន៍
  • ជ្រើសរើសថ្នាំប្រឆាំងជំងឺផ្អែកលើបាក់តេរីដែលបានរកឃើញ
  • ដោះស្រាយប្រភពបាក់តេរីបរិស្ថានដូចជា E. coli ឬ Strep spp. ដើម្បីជៀសវាងការរលត់ឡើងវិញ

ប្រសិនបើគោបង្ហាញចំនួនកោសិកាសូម៉ាទិកខ្ពស់ជាញឹកញាប់ គួរពិចារណា៖ - កាត់ចេញគោដែលងាយរងជំងឺតាមបន្ទុកពូជ - កែលម្អអនាម័យ និងអាហារូបត្ថម្ភ

● ព័ត៌មានបន្ថែមសម្រាប់ការការពារ៖

  • រក្សាបរិស្ថានក្នុងផ្ទះ និងក្រៅផ្ទះឱ្យស្អាត
  • ផ្តល់អាហារត្រឹមត្រូវ និងជៀសវាងអាហារចាស់
  • កាត់សក់កន្ទុយ និងសក់ទឹកដោះ រក្សាសុខភាពជើង
  • ប្រើស្រោមដៃ កន្សែងប្រើម្តង និងសារធាតុសំអាតពេលបូមទឹកដោះ
  • រក្សាគោឱ្យឈររយៈពេល ៣០នាទីបន្ទាប់ពីបូមទឹកដោះ ដើម្បីឱ្យបិទបំពង់ស្វាស
