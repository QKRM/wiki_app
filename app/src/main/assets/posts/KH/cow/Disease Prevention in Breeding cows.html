ការការពារជំងឺក្នុងការបង្កាត់ពូជគោ

▣ 1. ការការពារជំងឺបន្តពូជ

● វិធានការបង្ការសំខាន់ៗ៖

  • ការចិញ្ចឹម និងការគ្រប់គ្រងអាហារូបត្ថម្ភ បានត្រឹមត្រូវ។
  • ការធ្វើលំហាត់ប្រាណឱ្យបានគ្រប់គ្រាន់ ដើម្បីការពារទាំងកង្វះអាហារូបត្ថម្ភ និងការធាត់
  • ការគ្រប់គ្រងអនាម័យ មុន និងក្រោយពេលសម្រាលកូន៖
  • ការពារការចម្លងរោគតាមរយៈទ្វាមាស
  • គ្រប់គ្រងសុកដែលបានរក្សាទុកភ្លាមៗ
  • តាមដាន និងព្យាបាលការហូរទឹករំអិលរយៈពេល 2-3 សប្តាហ៍ក្រោយសម្រាល

គោលដៅបន្តពូជចុងក្រោយគួរតែជា កូនគោមួយក្បាលក្នុងមួយឆ្នាំ ។

● ជំងឺបន្តពូជទូទៅ៖

1. មុខងារក្រពេញអូវែ

  • រួម​បញ្ចូល​ទាំង​ការ​អភិវឌ្ឍ​មិន​ទាន់​ដំណើរការ ការ​តំរែតំរង់ ឬ​ការ​ខូច​នៃ​អូវែរ
  • បណ្តាលមកពីការអភិវឌ្ឍ follicular ខ្សោយ
  • ការការពារ៖ ផ្តល់ចំណីគុណភាពខ្ពស់ ប្រូតេអ៊ីន ផូស្វ័រ និងវីតាមីនអា
  • ការព្យាបាល៖ ការចាក់បញ្ចូលតាមសាច់ដុំ PMSG ឬ HCG (500-1,000μg)

2. ជំងឺអូវុល

  • រួមបញ្ចូល estrus ជាប់លាប់ និង cysts ovarian
  • បណ្តាលមកពីអតុល្យភាពនៃ LH និង FSH
  • ការព្យាបាល៖ ការចាក់បញ្ចូលតាមសាច់ដុំនៃ HCG
  • ផ្អែកលើប្រេង: 5,000-10,000 IU
  • ផ្អែកលើទឹក: 10,000 IU

3. បង្កាត់ពូជគោម្តងទៀត

  • សរីរាង្គបន្តពូជធម្មតា និងវដ្ត ប៉ុន្តែមិនអាចមានគភ៌បន្ទាប់ពីការព្យាយាម 3+
  • គ្មានការព្យាបាលជាក់លាក់; ទាមទារការរួមផ្សំនៃអាហាររូបត្ថម្ភ និងអនាម័យត្រឹមត្រូវ។

4. ជំងឺ endometritis

  • បង្កឡើងដោយ Vibrio , Brucella , Trichomonas ឬ E. coli
  • លទ្ធផលមានខ្ទុះក្នុងស្បូន
  • ការព្យាបាល៖ ការស្រោចស្រពស្បូន និងការបញ្ចូលអង់ទីប៊ីយ៉ូទិកក្នុងស្បូន

▣ 2. ការគ្រប់គ្រងសុខភាព Udder

ការ​ថែរក្សា​សុខភាព​ស្បូន​គឺ​ចាំបាច់​សម្រាប់​ផលិតភាព​សុខុមាលភាព​គោ និង​ការ​រក​ប្រាក់​ចំណេញ​ពី​កសិដ្ឋាន។ សុខភាពស្បូនខ្សោយនាំឱ្យ៖ - កាត់បន្ថយទិន្នផលទឹកដោះគោ - គុណភាពទឹកដោះគោកាន់តែយ៉ាប់យ៉ឺន - គ្រុនក្តៅ បាត់បង់ចំណង់អាហារ និងអាចស្លាប់។

● ប្រភេទ និងរោគសញ្ញានៃជំងឺ mastitis៖

1. រោគ Mastitis គ្លីនិក

  • ពងបែក ក្តៅ និងឈឺចាប់
  • ការផ្លាស់ប្តូរដែលអាចមើលឃើញនៅក្នុងទឹកដោះគោ (កំណក, ខ្ទុះ)
  • ករណីធ្ងន់ធ្ងរអាចទាមទារការកាត់

2. ជំងឺរលាកស្រោមខួរ subclinical

  • មិនមានរោគសញ្ញាដែលអាចមើលឃើញទេ។
  • រកឃើញតែតាមរយៈការធ្វើតេស្តមន្ទីរពិសោធន៍ (CMT, វប្បធម៌បាក់តេរី)
  • ប្រសិនបើ​មិន​បាន​ព្យាបាល​ទេ ច្រើន​តែ​វិវត្ត​ទៅ​ជា​ជំងឺ​រលាក​សួត​តាម​គ្លីនិក ហើយ​បណ្តាល​ឱ្យ​ខាតបង់​សេដ្ឋកិច្ច

● កត្តាហានិភ័យសម្រាប់ពងបែកមិនប្រក្រតី៖

  1. ពិការភាពកាយវិភាគសាស្ត្រ ៖
  2. សាច់ដុំ​ដើម​សុដន់​ខ្សោយ រូបរាង​សុដន់​ខុស​ប្រក្រតី ឬ​ប្រឡាយ
  3. ផ្ទះគ្មានអនាម័យ ៖
  4. ជង្រុកកខ្វក់ ខ្វះអនាម័យ ពូកមិនស្អាត
  5. អតុល្យភាពអាហារូបត្ថម្ភ ៖
  6. កង្វះសារធាតុរ៉ែ/វីតាមីន ការផ្លាស់ប្តូរចំណីភ្លាមៗ ចំណីផ្សិត
  7. បញ្ហាអ័រម៉ូន/មេតាបូលីស ៖
  8. អរម៉ូនអ៊ឹស្ត្រូសែនលើសឬកង្វះប្រូសេស្តេរ៉ូន
  9. ការថែទាំមុន/ក្រោយការសម្រាលកូនដោយមិនមានអនាម័យ ៖
  10. គ្មានការព្យាបាលដោយថ្នាំអង់ទីប៊ីយោទិចស្ងួត គ្មានការជ្រលក់ទឹកដោះ
  11. បញ្ហាម៉ាស៊ីនទឹកដោះគោ ៖
  12. សម្ពាធបូមធូលីខ្ពស់ ស្រទាប់ពាក់ អត្រាទឹកដោះគោលឿន (> 65 ចង្វាក់/នាទី)
  13. អនាម័យទឹកដោះគោមិនល្អ ៖
  14. កន្សែងរួម គ្មានអនាម័យ ប្រដាប់ភេទ
  15. ទំនោរហ្សែន ៖
  16. ការបង្កាត់ពូជសត្វគោជាមួយនឹងភាពងាយនឹងកើតជំងឺ mastitis

▣ 3. វិធានការការពារជំងឺពុកឆ្អឹង

● បច្ចេកទេសទឹកដោះគោត្រឹមត្រូវ និងការត្រួតពិនិត្យម៉ាស៊ីន

  • ពិនិត្យឱ្យបានទៀងទាត់៖
  • សម្ពាធសុញ្ញកាស (ល្អបំផុត: ~50 kPa)
  • ល្បឿនទឹកដោះគោ
  • ការជំនួសស្រទាប់
  • សមត្ថភាពបូមធូលី

ការអនុវត្តសំខាន់ៗ៖

  1. សម្អាត​តែ​ក្បាល​សុដន់​ប៉ុណ្ណោះ (មិន​មែន​ពង​ទា​ទាំងមូល)
  2. ការ​ច្រូត​ទឹក​ដោះ​មុន​ពេល​ចេញ​ទឹក ៖
  3. រំញោចការបញ្ចេញអុកស៊ីតូស៊ីន
  4. ជួយរកឃើញជំងឺ mastitis គ្លីនិកដំបូង
  5. រារាំងទឹកដោះគោដែលមានជាតិអង់ទីប៊ីយ៉ូទិកពីការលាយបញ្ចូលគ្នា

⚠️ ប្រើ ពែងឆ្នូត (មិនមែនកំរាលឥដ្ឋ) ដើម្បីជៀសវាងការបំពុលបរិស្ថាន

  1. ជៀសវាងការបំបៅទឹកដោះច្រើន ៖
  2. ទឹកដោះគោមិនគួរលើសពី 5 នាទី។
  3. ប្រើឯកតាទឹកដោះគោអតិបរមា 3 ក្នុងមនុស្សម្នាក់

  4. ជៀសវាងការច្រូតម៉ាស៊ីនបន្ទាប់ពីលំហូរធម្មជាតិបានបញ្ចប់

  5. គោ​ដែល​ឆ្លង​ដោយ​ទឹកដោះ​មាន​រយៈពេល​យូរ និង​ដាក់​វា​ឱ្យ​នៅ​ដាច់ដោយឡែក ដើម្បី​កាត់បន្ថយ​ការរីករាលដាល​ជំងឺ

● ការជ្រលក់ទឹកដោះ និងបាញ់ថ្នាំសម្លាប់មេរោគ

  • បន្ទាប់ពីការបំបៅទឹកដោះ ប្រឡាយទឹកដោះនៅតែបើករយៈពេល 15-30 នាទី ដែលប្រថុយនឹងបាក់តេរី
  • ការសម្លាប់មេរោគគឺចាំបាច់ដោយប្រើ៖
  • Teat Dips : ជំនួសដំណោះស្រាយឱ្យបានញឹកញាប់ដើម្បីរក្សាប្រសិទ្ធភាព
  • ការបាញ់ថ្នាំ ៖ ធានាបាននូវការគ្របដណ្តប់ពេញលេញនៃផ្ទៃដើម

● ការគ្រប់គ្រងរយៈពេលស្ងួត

ការព្យាបាលគោស្ងួតមានសារៈសំខាន់ណាស់សម្រាប់សុខភាពស្បូន៖ - លាបម្សៅគោស្ងួតដែលមានសកម្មភាពយូរ ឬថ្នាំអង់ទីប៊ីយោទិចខាងក្នុង - ការពារជំងឺ mastitis អំឡុងពេលស្ងួត និងការបំបៅដោះកូនដំបូង។

សកម្មភាពសំខាន់ៗ៖

  • ប្រមូលសំណាកទឹកដោះគោតាមត្រីមាសសម្រាប់ការធ្វើតេស្តមន្ទីរពិសោធន៍
  • ជ្រើសរើសថ្នាំអង់ទីប៊ីយោទិចដោយផ្អែកលើភ្នាក់ងារបង្ករោគ
  • ដោះស្រាយប្រភពបរិស្ថាននៃ E. coli ឬ Strep spp ។ ដើម្បីជៀសវាងការកើតឡើងវិញ។

ប្រសិនបើគោបង្ហាញចំនួនកោសិកា somatic ខ្ពស់ (SCC) ម្តងហើយម្តងទៀត សូមពិចារណា៖ - សម្លាប់សត្វគោដែលងាយទទួលហ្សែន - ការកែលម្អអនាម័យ និងអាហារូបត្ថម្ភ

● គន្លឹះបង្ការបន្ថែម៖

  • ថែរក្សា បរិស្ថានក្នុងផ្ទះ និងខាងក្រៅឱ្យស្អាត
  • ផ្តល់ អាហារដែលមានតុល្យភាព និងជៀសវាងចំណីផ្សិត
  • កាត់កន្ទុយ និងសក់ udder រក្សាសុខភាពជើង
  • ប្រើស្រោមដៃ កន្សែងដែលអាចចោលបាន និងថ្នាំសំលាប់មេរោគអំឡុងពេលបំបៅដោះ
